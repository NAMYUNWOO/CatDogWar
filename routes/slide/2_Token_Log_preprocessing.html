def mean_except_outlier(series):
    q1 = series.quantile(0.25)
    q3 = series.quantile(0.75)
    iqr = q3 - q1 
    series2 = series[series > q1 - N_IQR*iqr]
    if len(series2) == 0:
        return np.mean(series)
    return np.mean(series2)
